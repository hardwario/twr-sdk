538 {

541  for (crc16 = 0xffff; length != 0; length--, buffer++)
542  {
543  crc16 ^= *buffer;
544 
545  for (int i = 0; i < 8; i++)
546  {
547  if ((crc16 & 1) != 0)
548  {
549  crc16 >>= 1;
